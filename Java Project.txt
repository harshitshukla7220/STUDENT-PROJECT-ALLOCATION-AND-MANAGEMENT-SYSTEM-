import java.io.*;
import java.util.ArrayList;

public class AllocationManager {
Â    private ArrayList<Student> students;
Â    private final String FILE_NAME = "data.txt";

Â    public AllocationManager() {
Â        students = new ArrayList<>();
Â        loadData();
Â    }

Â    public void addAllocation(String studentName, String projectTitle) {
Â        Student s = new Student(studentName, projectTitle, "Pending");
Â        students.add(s);
Â        saveData();
Â    }

Â    public ArrayList<Student> getAllocations() {
Â        return students;
Â    }

Â    public void updateStatus(int index, String newStatus) {
Â        if (index >= 0 && index < students.size()) {
Â            students.get(index).setStatus(newStatus);
Â            saveData();
Â        }
Â    }

Â    private void loadData() {
Â        File file = new File(FILE_NAME);
Â        if (!file.exists()) return;

Â        try (BufferedReader br = new BufferedReader(new FileReader(file))) {
Â            String line;
Â            while ((line = br.readLine()) != null) {
Â                String[] parts = line.split(",", -1);
Â                if (parts.length == 3) {
Â                    students.add(new Student(parts[0], parts[1], parts[2]));
Â                }
Â            }
Â        } catch (IOException e) {
Â            System.out.println("Error reading file: " + e.getMessage());
Â        }
Â    }

Â    private void saveData() {
Â        try (BufferedWriter bw = new BufferedWriter(new FileWriter(FILE_NAME))) {
Â            for (Student s : students) {
Â                bw.write(s.toString());
Â                bw.newLine();
Â            }
Â        } catch (IOException e) {
Â            System.out.println("Error saving file: " + e.getMessage());
Â        }
Â    }
